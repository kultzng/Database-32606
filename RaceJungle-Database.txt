CREATE VIEW HGDM24_Customerdetails AS
SELECT c.customerName, c.phoneNumber, c.email, c.gender, re.transactionID, r.raceName 
FROM Customer c, Register re, Race r
WHERE re.raceID = r.raceID
AND r.raceID = 'HGDM24'
AND re.customerID = c.customerID
ORDER BY customerName;
